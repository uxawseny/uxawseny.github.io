    protected synchronized Class<?> loadClass(String name, boolean resolve)
throws ClassNotFoundException {
    // йҰ–е…Ҳжүҫзј“еӯҳжҳҜеҗҰжңү class
    Class c = findLoadedClass(name);
    if (c == null) {
        //жІЎжңүеҲӨж–ӯжңүжІЎжңүзҲ¶зұ»
        try {
                if (parent != null) {
                    //жңүзҡ„иҜқпјҢз”ЁзҲ¶зұ»йҖ’еҪ’иҺ·еҸ– class
                    c = parent.loadClass(name, false);
                } else {
                    //жІЎжңүзҲ¶зұ»гҖӮйҖҡиҝҮиҝҷдёӘж–№жі•жқҘеҠ иҪҪ
                    c = findBootstrapClassOrNull(name);
                }
        } catch (ClassNotFoundException e) {
                // ClassNotFoundException thrown if class not found
                // from the non-null parent class loader
        }
        if (c == null) {
            // еҰӮжһңиҝҳжҳҜжІЎжңүжүҫеҲ°пјҢи°ғз”Ё findClass(name)еҺ»жүҫиҝҷдёӘзұ»
            c = findClass(name);
        }
    }
    if (resolve) {
        resolveClass(c);
    }
    return c;
}

package com.imux.wxsell.service;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.data.redis.core.StringRedisTemplate;
import org.springframework.stereotype.Component;
import org.springframework.util.StringUtils;

/**
 * redis lock
 */
@Component
@Slf4j
public class RedisLock {
    @Autowired
    private StringRedisTemplate stringRedisTemplate;

    /**
     * еҠ й”Ғ
     *
     * @param key   productId - е•Ҷе“Ғзҡ„е”ҜдёҖж Үеҝ—
     * @param value еҪ“еүҚж—¶й—ҙ+и¶…ж—¶ж—¶й—ҙ>>>еҚіиҝҮжңҹзҡ„ж—¶еҲ»
     * @return
     */
    public boolean lock(String key, String value) {
        //setIfAbsent >>>>>>>redisзҡ„SETNXж–№жі•
        if (stringRedisTemplate.opsForValue().setIfAbsent(key, value)) {//еҜ№еә”setnxе‘Ҫд»Ө
            //еҸҜд»ҘжҲҗеҠҹи®ҫзҪ®,д№ҹе°ұжҳҜkeyдёҚеӯҳеңЁ
            return true;
        }

        //еҲӨж–ӯй”Ғи¶…ж—¶ - йҳІжӯўеҺҹжқҘзҡ„ж“ҚдҪңејӮеёёпјҢжІЎжңүиҝҗиЎҢи§Јй”Ғж“ҚдҪң  йҳІжӯўжӯ»й”Ғ
        String currentValue = stringRedisTemplate.opsForValue().get(key);

        //еҰӮжһңй”ҒиҝҮжңҹ
        if (!StringUtils.isEmpty(currentValue) &&
                Long.parseLong(currentValue) < System.currentTimeMillis()) {//currentValueдёҚдёәз©әдё”е°ҸдәҺеҪ“еүҚж—¶й—ҙ
            //иҺ·еҸ–дёҠдёҖдёӘй”Ғзҡ„ж—¶й—ҙvalue
            String oldValue = stringRedisTemplate.opsForValue().getAndSet(key, value);//еҜ№еә”getsetпјҢеҰӮжһңkeyеӯҳеңЁ

            //еҒҮи®ҫдёӨдёӘзәҝзЁӢеҗҢж—¶иҝӣжқҘпјҢkeyиў«еҚ з”ЁдәҶгҖӮиҺ·еҸ–зҡ„еҖјcurrentValue=A(getеҸ–зҡ„ж—§зҡ„еҖјиӮҜе®ҡжҳҜдёҖж ·зҡ„),дёӨдёӘзәҝзЁӢзҡ„valueйғҪжҳҜB,keyйғҪжҳҜK.й”Ғж—¶й—ҙе·Із»ҸиҝҮжңҹдәҶгҖӮ
            //иҖҢиҝҷйҮҢйқўзҡ„getAndSetдёҖж¬ЎеҸӘдјҡдёҖдёӘжү§иЎҢпјҢд№ҹе°ұжҳҜдёҖдёӘжү§иЎҢд№ӢеҗҺпјҢдёҠдёҖдёӘзҡ„valueе·Із»ҸеҸҳжҲҗдәҶBгҖӮеҸӘжңүдёҖдёӘзәҝзЁӢиҺ·еҸ–зҡ„дёҠдёҖдёӘеҖјдјҡжҳҜAпјҢеҸҰдёҖдёӘзәҝзЁӢжӢҝеҲ°зҡ„еҖјжҳҜBгҖӮ
            if (!StringUtils.isEmpty(oldValue) && oldValue.equals(currentValue)) {
                //oldValueдёҚдёәз©әдё”oldValueзӯүдәҺcurrentValueпјҢд№ҹе°ұжҳҜж ЎйӘҢжҳҜдёҚжҳҜдёҠдёӘеҜ№еә”зҡ„е•Ҷе“Ғж—¶й—ҙжҲіпјҢд№ҹжҳҜйҳІжӯўе№¶еҸ‘
                return true;
            }
        }
        return false;
    }

    /**
     * и§Јй”Ғ
     *
     * @param key
     * @param value
     */
    public void unlock(String key, String value) {
        try {
            String currentValue = stringRedisTemplate.opsForValue().get(key);
            if (!StringUtils.isEmpty(currentValue) && currentValue.equals(value)) {
                stringRedisTemplate.opsForValue().getOperations().delete(key);//еҲ йҷӨkey
            }
        } catch (Exception e) {
            log.error("[RedisеҲҶеёғејҸй”Ғ] и§Јй”ҒеҮәзҺ°ејӮеёёдәҶпјҢ{}", e);
        }
    }
}
